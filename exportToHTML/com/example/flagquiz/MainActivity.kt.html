81       private fun randomBtn(countryName: String?): ArrayList<Button> {
82           val array = ArrayList<Button>()
83           val arrayText = ArrayList<String>()
84   
85           for (c in countryName!!) {
86               arrayText.add(c.toString())//uzbekistan  = u , z , b , e , k , i , s , t , a , n
87           }
88           if (arrayText.size != 12) {
89               val str = "ABCDEFGHIJKLMNOPQRSTUVXYZ"
90               for (i in arrayText.size until 12) {
91                   val random = Random().nextInt(str.length)
92                   arrayText.add(str[random].toString())
93               }
94           }
95           arrayText.shuffle()
96   
97           for (i in 0 until arrayText.size) {
98               val button = Button(this)
99               customBtn(button,15.0,"#D81E88E5","#FFFFFF","#000000","#FA0000",5.0)
100              button.layoutParams = LinearLayout.LayoutParams(LinearLayout.LayoutParams.WRAP_CONTENT, LinearLayout.LayoutParams.WRAP_CONTENT, 1.0f)
101              button.text = arrayText[i]
102              button.setOnClickListener(this)
103              array.add(button)
104          }
105          return array
106      }

108      override fun onClick(v: View?) {
109          val button1 = v as Button
110          customBtn(button1,15.0,"#D81E88E5","#FFFFFF","#000000","#FA0000",5.0)
111          if (buttonArrayList.contains(button1)) {
112              lin_1_matn.removeView(button1)
113              var hasC = false
114              lin_2_btn1.children.forEach { button ->
115                  if ((button as Button).text.toString() == button1.text.toString()) {
116                      button.visibility = View.VISIBLE
117                      countryName = countryName.substring(0, countryName.length - 1)
118                      hasC = true
119                  }
120              }
121  
122              lin3_btn_2.children.forEach { button ->
123  
124                  if ((button as Button).text.toString() == button1.text.toString()) {
125  
126                      button.visibility = View.VISIBLE
127                      if (!hasC) {
128                          countryName = countryName.substring(0, countryName.length - 1)
129                      }
130                  }
131              }
132          } else {
133              button1.visibility = View.INVISIBLE
134              if (button1.text.toString().lowercase(Locale.getDefault()) == "a" && pressed) {
135                  Toast.makeText(this, "A harfi bosildi", Toast.LENGTH_SHORT).show()
136              }
137              countryName += button1.text.toString().uppercase(Locale.getDefault())
138              val button2 = Button(this)
139              customBtn(button2,15.0,"#D81E88E5","#FFFFFF","#000000","#FA0000",5.0)
140              button2.layoutParams = LinearLayout.LayoutParams(
141                  LinearLayout.LayoutParams.WRAP_CONTENT,
142                  LinearLayout.LayoutParams.WRAP_CONTENT,
143                  1.0f
144              )
145              button2.text = button1.text
146  
147              button2.setOnClickListener(this)
148              buttonArrayList.add(button2)
149              lin_1_matn.addView(button2)
150              matnTogri()
151          }
152      }

154      private fun matnTogri() {
155          if (countryName == flagArrayList[count].name?.uppercase(Locale.getDefault())) {
156              Toast.makeText(this@MainActivity, "Successful", Toast.LENGTH_SHORT).show()
157              if (count == flagArrayList.size - 1) {
158                  count = 0
159              } else {
160                  count++
161              }
162              btnJoylaCount()
163          } else {
164              if (countryName.length == flagArrayList[count].name?.length) {
165                  Toast.makeText(this@MainActivity, "Error", Toast.LENGTH_SHORT).show()
166                  lin_1_matn.removeAllViews()
167                  lin3_btn_2.removeAllViews()
168                  lin_2_btn1.removeAllViews()
169                  btnJoyla(flagArrayList[count].name)
170                  countryName = ""
171              }
172          }
173      }
174      private fun customBtn(view:View,radius:Double,a:String,b:String,c:String,d:String,s:Double){
175          val style = GradientDrawable()
176          style.cornerRadius = radius.toFloat()
177          if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) {
178              style.color =
179                  ColorStateList(
180                      arrayOf(
181                          intArrayOf(-android.R.attr.state_pressed),
182                          intArrayOf(android.R.attr.state_pressed)
183                      ), intArrayOf(Color.parseColor(a), Color.parseColor(b))
184                  )
185          }
186          if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) {
187              style.setStroke(
188                  s.toInt(),
189                  ColorStateList(
190                      arrayOf(
191                          intArrayOf(-android.R.attr.state_pressed),
192                          intArrayOf(android.R.attr.state_pressed)
193                      ), intArrayOf(Color.parseColor(c), Color.parseColor(d))
194                  )
195              )
196          }
197          view.background = style
198      }
